\problema{ }\label{pro:1}
	Muestre que:
	\begin{enumerate}[\hspace{1cm}(a)]
		\item $7|21$.
		\item $3|21$.
		\item $14|42$.
		\item $6|42$.
	\end{enumerate}
	\solution 
	\part
	Haciendo un procedimiento clásico de división:\\
	\begin{center}
		\intlongdivision[german division sign = $\,\div\,$]{21}{7}
	\end{center}
	Obtenemos que el resto es $5$ por lo tanto $7$ si divide a $21$\QEPD
	\part
	Haciendo un procedimiento clásico de división:\\
	\begin{center}
		\intlongdivision[german division sign = $\,\div\,$]{21}{3}
	\end{center}
	Obtenemos que el resto es $5$ por lo tanto $3$ si divide a $21$\QEPD
	\part
	Haciendo un procedimiento clásico de división:\\
	\begin{center}
		\intlongdivision[german division sign = $\,\div\,$]{42}{14}
	\end{center}
	Obtenemos que el resto es $0$ por lo tanto $14$ si divide a $42$\QEPD
	\part
	Haciendo un procedimiento clásico de división:\\
	\begin{center}
		\intlongdivision[german division sign = $\,\div\,$]{42}{6}
	\end{center}
	Obtenemos que el resto es $0$ por lo tanto $6$ si divide a $42$\QEPD
